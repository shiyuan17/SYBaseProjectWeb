upstream sybase_auth_center_upstream {
    server 127.0.0.1:8081;
    keepalive 16;
}

upstream sybase_bl_center_upstream {
    server 127.0.0.1:8080;
    keepalive 32;
}

server {
    listen 80;
    server_name web-ele.local;

    # set this to the real dist path on the server
    root /data/www/sybase/web-ele/dist;
    index index.html;

    client_max_body_size 100m;

    location = /health {
        access_log off;
        add_header Content-Type text/plain;
        return 200 "ok\n";
    }

    location /api/v1/auth/ {
        proxy_pass http://sybase_auth_center_upstream;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header Connection "";
        proxy_read_timeout 300s;
        proxy_send_timeout 300s;
    }

    location /api/v1/ {
        proxy_pass http://sybase_bl_center_upstream;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header Connection "";
        proxy_read_timeout 300s;
        proxy_send_timeout 300s;
    }

    location / {
        try_files $uri $uri/ /index.html;
    }
}
